RGB vs HSB

加法混色・減法混色

  • 引数の並び:(red, green, blue, alpha)
    • 引数１つならグレースケール
    • 引数３つなら不透明
  • 明示的にもかける:(red)
  • x6でもかける:('#ff0000')
function setup() {
  createCanvas(720, 720);
  noCursor();
}

function draw() {
  colorMode(HSB, 360, 100, 100);
  rectMode(CENTER);
  noStroke();
  background(mouseY/2, 100, 100);

  fill(360-mouseY/2, 100, 100);
  rect(360, 360, mouseX+1, mouseX+1);
}

変数

  • 宣言をしてから使う
  • jsの場合は
    • let
    • const
  • 明示的に名前をつける
  • 器のようなものでなんでも入る
  • 出し入れ可能
//1,宣言：使用する変数名を宣言（箱を準備する）
let num;
//2,代入：変数の箱に値を入れる
//この場合のイコールは代入を意味する
num = 400;
//3,演算して代入
//等式ではない！代入！
num = num + 1;
//下でも同じ意味
num++;

比較演算子

右辺と左辺を比較する

  • 等価：==
  • 不等価：!=
  • 一致：===
  • 不一致：!==

関係演算子

  • 大なり：>
  • 大なりイコール：>=
  • 小なり：<
  • 小なりイコール：<=

論理演算子

  • AND：&&
  • OR：||
  • NOT：!

分岐（if文）

条件式を書き、そこに当てはまればカーリーブラケット内部の処理を実行する

if (i == 0) {
  text("iは0です", 0, 0);
}else if (i > 0) {
  text("iは正の数です", 0, 0);
}else {
  text("iは負の数です", 0, 0);
}

反復(for文)

初期化式と継続条件式と再初期化式で成り立つ

for (let i = 0; i < 10; i++) {
  text("iは10未満です");
}

無限ループ(eternal loop)

for (let i = 0; i < 10; i--) {
  text("iは10未満です");
}

let originX = 0,
    originY = 0;

function setup() {
  createCanvas(600, 600);
  noCursor();
  print('setOrigin(X:' + originX + ',Y:' + originY + ')')
}

function draw() {
  background(220);
  const step = 50;
  stroke(200);
  for (let x = step; x < width; x += step) {
    line(x, 0, x, height);
  }

  for (let y = step; y < height; y += step) {
    line(0, y, width, y);
  }

  stroke(255, 0, 255);
  point(originX, originY);
  noStroke();
  fill(255, 0, 255);
  push();
  text('ORIGIN', originX - 44, originY - 2);

  if (mouseX > 0 && mouseX <= width && mouseY > 0 && mouseY <= height) {
    pop();
    ellipse(mouseX, mouseY, 5, 5);
    let rerativeX = mouseX - originX,
        rerativeY = mouseY - originY;
    text('(x:' + rerativeX + ',y:' + rerativeY + ')', mouseX + 5, mouseY + 10);
    if (mouseX == originX || mouseY == originY) {
      stroke('#FF00FF');
      line(originX, 0, originX, height);
      line(0, originY, width, originY);
    }
    }else {
      stroke('#FF00FF');
      rect(originX - 3, originY - 3, 6, 6);
      pop();
      text('(x:0,y:0)', originX + 5, originY + 10);
    }
  }

  function keyPressed() {
    if (key == 's') {
      originX = mouseX;
      originY = mouseY;
      print('setOrigin(X:' + originX + ',Y:' + originY + ')')
    }
    if (key == 'z') {
      originX = 0;
      originY = 0;
      print('setOrigin(X:' + originX + ',Y:' + originY + ')')
    }
  }
}
